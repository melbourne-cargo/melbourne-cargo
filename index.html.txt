<button onclick="calc()">Calculate</button>

<h3 id="result"></h3>

<script>
function calc() {
  let d = Number(document.getElementById('d').value);
  let r = Number(document.getElementById('r').value);
  let b = Number(document.getElementById('b').value);
  let w = Number(document.getElementById('w').value);
  let l = Number(document.getElementById('l').value);
  let wi = Number(document.getElementById('wi').value);
  let h = Number(document.getElementById('h').value);

  let volumetric = (l * wi * h) / 5000;
  let chargeable = Math.max(w, volumetric);

  let total = (d * r) + b + (chargeable * 0.6);

  document.getElementById('result').innerText = "Total Price: $" + total.toFixed(2);
}
</script>
